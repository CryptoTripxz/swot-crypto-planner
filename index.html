function buildCal(){

  const grid=document.getElementById('calGrid');if(!grid)return;

  grid.innerHTML='';

  const now=new Date();

  const todayY=now.getFullYear(),todayM=now.getMonth(),todayD=now.getDate();

  const firstDow=new Date(calViewYear,calViewMonth,1).getDay();

  const daysInMonth=new Date(calViewYear,calViewMonth+1,0).getDate();

  const monthName=MONTH_NAMES[calViewMonth];

  const lbl=document.getElementById('cal-month-label');if(lbl)lbl.textContent=`${monthName} ${calViewYear}`;

  let entryCount=0;

  for(let d=1;d<=daysInMonth;d++){if(dayEntries[fmtKey(calViewYear,calViewMonth,d)])entryCount++;}

  const cl=document.getElementById('cal-checkin-label');if(cl)cl.textContent=`Daily entries · ${entryCount} logged`;

  for(let i=0;i<firstDow;i++){const e=document.createElement('div');e.className='cday emp';grid.appendChild(e);}

  for(let d=1;d<=daysInMonth;d++){

    const cell=document.createElement('div');

    const dow=(firstDow+d-1)%7;

    const isToday=(d===todayD&&calViewMonth===todayM&&calViewYear===todayY);

    const key=fmtKey(calViewYear,calViewMonth,d);

    const hasEntry=!!dayEntries[key];

    const isPast=(new Date(calViewYear,calViewMonth,d))<new Date(todayY,todayM,todayD);

    let cls='cday';

    if(isToday)cls+=' td';

    else if(hasEntry)cls+=' ck';

    else if(dow===0)cls+=' sun';

    cell.className=cls;cell.textContent=d;

    if(hasEntry&&!isToday){const dot=document.createElement('div');dot.style.cssText='width:4px;height:4px;border-radius:50%;background:var(--teal);margin:1px auto 0;';cell.appendChild(dot);}

    cell.onclick=(()=>{const dk=key,dn=d,iP=isPast,hE=hasEntry,isT=isToday;return()=>{

      if(!isT&&!iP&&!hE){showToast(`⏳ ${monthName} ${dn} — upcoming`);return;}

      openDayModal(dk,`${monthName} ${dn}, ${calViewYear}`);

    };})();

    grid.appendChild(cell);

  }

  renderStreakDots();renderGoalsList();

}

function updateScore(id,val){

  const item=items.find(i=>i.id===id);if(!item)return;

  item.score=parseInt(val);const c=getScoreColor(item.score);const sw=getScoreWord(item.score);const pct=(item.score/10)*100;

  const upd=(elId,prop,v)=>{const el=document.getElementById(elId);if(el)el[prop]=v;};

  upd('snum-'+id,'textContent',item.score);upd('snum-'+id,'style.color',c);

  upd('swd-'+id,'textContent',sw.word);upd('swd-'+id,'style.background',sw.bg);upd('swd-'+id,'style.color',sw.color);

  const f=document.getElementById('sfill-'+id);

  if(f){f.style.width=pct+'%';f.style.background=`linear-gradient(90deg,${c},${c}cc)`;}

  const sl=document.getElementById('slider-'+id);if(sl)sl.style.color=c;

function stepScore(id,delta){

}
